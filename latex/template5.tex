\newcommand{\namesection}[3]{%
  \centering{%
    \fontsize{40pt}{48pt}\selectfont #1
    \selectfont #2
  }\mbox{}\\[5pt]
  \centering{\color{headings}\fontsize{11pt}{14pt}\selectfont #3}
  \noindent\makebox[\linewidth]{\rule{\paperwidth}{0.4pt}}
  \vspace{-15pt}
}

% Education entry (left column)
% \education{Institution}{Degree}{Location}{Details line 1}{Details line 2 (optional)}
\newcommand{\education}[5]{%
  \AppendTo{\EducationList}{%
    \subsection{#1}%
    \descript{#2}%
    \location{#3}%
    #4\\
    #5\\
    \sectionsep
  }%
}

% Links (left column)
% \linkitem{Label}{URL}{DisplayText}
\newcommand{\linkitem}[3]{%
  \AppendTo{\LinksList}{%
    #1://~\href{#2}{\custombold{#3}}\\
  }%
}

% Experience (right column)
% \experience{Org}{Role}{Dates | Location}{items = \item ...}
\newcommand{\experience}[4]{%
  \AppendTo{\ExperienceList}{%
    \runsubsection{#1}%
    \descript{| #2}%
    \location{#3}%
    \vspace{\topsep}%
    \begin{tightemize}
      #4
    \end{tightemize}
    \sectionsep
  }%
}
